68 void calcResize()
69 {
70 // calculate bounding box and store internally
71 glm::vec3 mins, maxs;
72 mins = glm::vec3(std::numeric_limits<float>::max());
73 maxs = glm::vec3(std::numeric_limits<float>::min());
74 for(auto mesh : this->my_mesh.getMeshes())
75 {
76 mins = glm::min(mins, mesh->min);
77 maxs = glm::max(maxs, mesh->max);
78 }
79
80 glm::vec3 mid = maxs * 0.5f + mins * 0.5f;
81 this->mid = mid;
82 glm::vec3 temp = maxs - mins;
83 float largest_extent = glm::max(glm::max(temp.x, temp.y), temp.z);
84 largest_extent = 2.0f / largest_extent;
85
86 // We need to:
87 // Translate -mid direction
88 // Scale down by 2/largest extent (handled above to save cycles)
89 this->resizeMat = glm::scale(glm::mat4(1.0f), glm::vec3(largest_extent, largest_extent, largest_extent));
90 this->resizeMat = this->resizeMat * glm::translate(glm::mat4(1.0f), mid * -1.0f);
91
92 // lastly, convert mid from obj coordinates (space) to world coordinates (since we always multiply by our resize anyways)
93 this->mid = glm::vec3(this->resizeMat * glm::vec4(this->mid, 1.0));
94 }

151 void setModel(Program & curS, glm::vec3 trans, float rotY, float rotX, float sc) {
152 glm::mat4 Trans = glm::translate( glm::mat4(1.0f), trans);
153 glm::mat4 RotX = glm::rotate( glm::mat4(1.0f), rotX, glm::vec3(1, 0, 0));
154 glm::mat4 RotY = glm::rotate( glm::mat4(1.0f), rotY, glm::vec3(0, 1, 0));
155 glm::mat4 ScaleS = glm::scale(glm::mat4(1.0f), glm::vec3(sc));
156 glm::mat4 ctm = Trans*RotX*RotY*ScaleS;
157 glUniformMatrix4fv(curS.getUniform("M"), 1, GL_FALSE, value_ptr(ctm));
